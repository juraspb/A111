20  * @brief Example that shows how to use the sparse service
21  *
22  * This is an example on how the sparse service can be used.
23  * The example executes as follows:
24  * - Activate Radar System Software (RSS)
25  * - Create a sparse service configuration
26  * - Create a sparse service using the previously created configuration
27  * - Activate the sparse service
28  * - Get the result and print it 5 times
29  * - Deactivate and destroy the sparse service
30  * - Destroy the sparse service configuration
31  * - Deactivate Radar System Software (RSS)
32  */

86 bool execute_sparse(acc_service_configuration_t sparse_configuration)
87 {
88  acc_service_handle_t handle = acc_service_create(sparse_configuration);
89 
90  if (handle == NULL)
91  {
92  fprintf(stderr, "acc_service_create() failed\n");
93  return false;
94  }
95 
96  acc_service_sparse_metadata_t sparse_metadata;
97  acc_service_sparse_get_metadata(handle, &sparse_metadata);
98 
99  printf("Start: %d mm\n", (int)(sparse_metadata.start_m * 1000.0f));
100  printf("Length: %u mm\n", (unsigned int)(sparse_metadata.length_m * 1000.0f));
101  printf("Data length: %u\n", (unsigned int)(sparse_metadata.data_length));
102 
103  uint16_t sweeps_per_frame = acc_service_sparse_configuration_sweeps_per_frame_get(sparse_configuration);
104  uint16_t sweep_length = sparse_metadata.data_length / sweeps_per_frame;
105 
106  printf("Sweeps per frame: %u\n", (unsigned int)sweeps_per_frame);
107  printf("Sweep length: %u\n", (unsigned int)sweep_length);
108 
109  uint16_t data[sparse_metadata.data_length];
110  acc_service_sparse_result_info_t result_info;
111 
112  if (!acc_service_activate(handle))
113  {
114  printf("acc_service_activate() failed\n");
115  acc_service_destroy(&handle);
116  return false;
117  }
118 
119  bool success = true;
120  const int iterations = 5;
121 
122  for (int i = 0; i < iterations; i++)
123  {
124  success = acc_service_sparse_get_next(handle, data, sparse_metadata.data_length, &result_info);
125 
126  if (!success)
127  {
128  fprintf(stderr, "acc_service_sparse_get_next() failed\n");
129  break;
130  }
131 
132  printf("Sparse data:\n");
133 
134  for (uint16_t sweep = 0; sweep < sweeps_per_frame; sweep++)
135  {
136  printf("Sweep %u:\t", (unsigned int)(sweep + 1));
137 
138  for (uint16_t sweep_idx = 0; sweep_idx < sweep_length; sweep_idx++)
139  {
140  printf("%6u", (unsigned int)data[(sweep * sweep_length) + sweep_idx]);
141  }
142 
143  printf("\n");
144  }
145 
146  printf("\n");
147  }
148 
149  bool deactivated = acc_service_deactivate(handle);
150 
151  acc_service_destroy(&handle);
152 
153  return deactivated && success;
154 }
