20 /**
21  * @brief Example that shows how to use the IQ service
22  *
23  * This is an example on how the IQ service can be used.
24  * The example executes as follows:
25  * - Activate Radar System Software (RSS)
26  * - Create an IQ service configuration
27  * - Create an IQ service using the previously created configuration
28  * - Activate the IQ service
29  * - Get the result and print it 5 times
30  * - Deactivate and destroy the IQ service
31  * - Destroy the IQ service configuration
32  * - Deactivate Radar System Software (RSS)
33  */

84 bool execute_iq(acc_service_configuration_t iq_configuration)
85 {
86  acc_service_handle_t handle = acc_service_create(iq_configuration);
87 
88  if (handle == NULL)
89  {
90  fprintf(stderr, "acc_service_create() failed\n");
91  return false;
92  }
93 
94  acc_service_iq_metadata_t iq_metadata;
95  acc_service_iq_get_metadata(handle, &iq_metadata);
96 
97  printf("Start: %d mm\n", (int)(iq_metadata.start_m * 1000.0f));
98  printf("Length: %u mm\n", (unsigned int)(iq_metadata.length_m * 1000.0f));
99  printf("Data length: %u\n", (unsigned int)(iq_metadata.data_length));
100 
101  float complex data[iq_metadata.data_length];
102  acc_service_iq_result_info_t result_info;
103 
104  if (!acc_service_activate(handle))
105  {
106  fprintf(stderr, "acc_service_activate() failed\n");
107  acc_service_destroy(&handle);
108  return false;
109  }
110 
111  bool success = true;
112  const int iterations = 5;
113 
114  for (int i = 0; i < iterations; i++)
115  {
116  success = acc_service_iq_get_next(handle, data, iq_metadata.data_length, &result_info);
117 
118  if (!success)
119  {
120  fprintf(stderr, "acc_service_iq_get_next() failed\n");
121  break;
122  }
123 
124  printf("IQ data in polar coordinates (r, phi) (multiplied by 1000):\n");
125 
126  for (uint16_t j = 0; j < iq_metadata.data_length; j++)
127  {
128  if ((j > 0) && ((j % 8) == 0))
129  {
130  printf("\n");
131  }
132 
133  printf("(%d, %d)\t", (int)(cabsf(data[j]) * 1000), (int)(cargf(data[j]) * 1000));
134  }
135 
136  printf("\n");
137  }
138 
139  bool deactivated = acc_service_deactivate(handle);
140 
141  acc_service_destroy(&handle);
142 
143  return deactivated && success;
144 }
