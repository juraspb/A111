19 /**
20  * @brief Example that shows how to use the Power Bins service
21  *
22  * This is an example on how the Power Bins service can be used.
23  * The example executes as follows:
24  * - Activate Radar System Software (RSS)
25  * - Create a Power Bins service configuration
26  * - Create a Power Bins service using the previously created configuration
27  * - Activate the Power Bins service
28  * - Get the result and print it 5 times
29  * - Deactivate and destroy the Power Bins service
30  * - Destroy the Power Bins service configuration
31  * - Deactivate Radar System Software (RSS)
32  */

82 bool execute_power_bins(acc_service_configuration_t power_bins_configuration)
83 {
84  acc_service_handle_t handle = acc_service_create(power_bins_configuration);
85 
86  if (handle == NULL)
87  {
88  fprintf(stderr, "acc_service_create() failed\n");

90  }
91 
92  acc_service_power_bins_metadata_t power_bins_metadata;
93  acc_service_power_bins_get_metadata(handle, &power_bins_metadata);
94 
95  printf("Start: %d mm\n", (int)(power_bins_metadata.start_m * 1000.0f));
96  printf("Length: %u mm\n", (unsigned int)(power_bins_metadata.length_m * 1000.0f));
97  printf("Bin count: %u\n", (unsigned int)(power_bins_metadata.bin_count));
98 
99  uint16_t data[power_bins_metadata.bin_count];
100 
101  acc_service_power_bins_result_info_t result_info;
102 
103  if (!acc_service_activate(handle))
104  {
105  printf("acc_service_activate() failed\n");
106  acc_service_destroy(&handle);
107  return false;
108  }
109 
110  bool success = true;
111  const int iterations = 5;
112 
113  for (int i = 0; i < iterations; i++)
114  {
115  success = acc_service_power_bins_get_next(handle, data, power_bins_metadata.bin_count, &result_info);
116 
117  if (!success)
118  {
119  fprintf(stderr, "acc_service_power_bins_get_next() failed\n");
120  break;
121  }
122 
123  printf("Power Bins data:\n");
124 
125  for (uint16_t j = 0; j < power_bins_metadata.bin_count; j++)
126  {
127  printf("%u\t", (unsigned int)(data[j]));
128  }
129 
130  printf("\n");
131  }
132 
133  bool deactivated = acc_service_deactivate(handle);
134 
135  acc_service_destroy(&handle);
136 
137  return deactivated && success;
138 }
