65 static void acc_hal_integration_sensor_transfer(acc_sensor_id_t sensor_id, uint8_t *buffer, size_t buffer_size)
66 {
67  (void)sensor_id; // Ignore parameter sensor_id
68 
69  HAL_GPIO_WritePin(A111_SPI_SS_GPIO_Port, A111_SPI_SS_Pin, GPIO_PIN_RESET);
70 
71  HAL_SPI_TransmitReceive(&A111_SPI_HANDLE, buffer, buffer, buffer_size, 5000);
72 
73  HAL_GPIO_WritePin(A111_SPI_SS_GPIO_Port, A111_SPI_SS_Pin, GPIO_PIN_SET);

77 static void acc_hal_integration_sensor_power_on(acc_sensor_id_t sensor_id)
78 {
79  (void)sensor_id; // Ignore parameter sensor_id
80 
81  HAL_GPIO_WritePin(A111_SPI_SS_GPIO_Port, A111_SPI_SS_Pin, GPIO_PIN_SET);
82 
83  // Enable PMU and wait
84  HAL_GPIO_WritePin(XC112_PMU_ENABLE_GPIO_Port, XC112_PMU_ENABLE_Pin, GPIO_PIN_SET);
85  acc_hal_integration_sleep_us(5000);
86 
87  HAL_GPIO_WritePin(XC112_ENABLE_N_GPIO_Port, XC112_ENABLE_N_Pin, GPIO_PIN_RESET);
88  acc_hal_integration_sleep_us(5000);
89 
90  HAL_GPIO_WritePin(A111_ENABLE_GPIO_Port, A111_ENABLE_Pin, GPIO_PIN_SET);
91  acc_hal_integration_sleep_us(5000);
92 
93  // Select communication with only Sensor 1 and deselect all other radar slots
94  HAL_GPIO_WritePin(XC112_SPI_S1_ENABLE_N_GPIO_Port, XC112_SPI_S1_ENABLE_N_Pin, GPIO_PIN_RESET);
95  HAL_GPIO_WritePin(XC112_SPI_S2_ENABLE_N_GPIO_Port, XC112_SPI_S2_ENABLE_N_Pin, GPIO_PIN_SET);
96  HAL_GPIO_WritePin(XC112_SPI_S3_ENABLE_N_GPIO_Port, XC112_SPI_S3_ENABLE_N_Pin, GPIO_PIN_SET);
97  HAL_GPIO_WritePin(XC112_SPI_S4_ENABLE_N_GPIO_Port, XC112_SPI_S4_ENABLE_N_Pin, GPIO_PIN_SET);
98 }

112 static bool acc_hal_integration_wait_for_sensor_interrupt(acc_sensor_id_t sensor_id, uint32_t timeout_ms)
113 {
114  (void)sensor_id; // Ignore parameter sensor_id
115 
116  const uint32_t wait_begin_ms = HAL_GetTick();
117  while (HAL_GPIO_ReadPin(A111_SENSOR_INTERRUPT_GPIO_Port, A111_SENSOR_INTERRUPT_Pin) != GPIO_PIN_SET)
118  {
119  if (timeout_ms == 0 || HAL_GetTick() - wait_begin_ms > timeout_ms)
120  {
121  return false;
122  }
123 
124  // Wait for the next tick interrupt
125  __WFI();
126  }
127 
128  return true;
129 }

144 static void acc_hal_integration_log(acc_log_level_t level, const char *module, const char *buffer)
145 {
146  uint32_t time_ms = acc_hal_integration_get_current_time();
147  char level_ch;
148 
149  unsigned int timestamp = time_ms;
150  unsigned int hours = timestamp / 1000 / 60 / 60;
151  unsigned int minutes = timestamp / 1000 / 60 % 60;
152  unsigned int seconds = timestamp / 1000 % 60;
153  unsigned int milliseconds = timestamp % 1000;
154 
155  level_ch = (level <= ACC_LOG_LEVEL_DEBUG) ? "EWIVD"[level] : '?';
156 
157  printf(LOG_FORMAT, hours, minutes, seconds, milliseconds, 0, level_ch, module, buffer);
158 }
