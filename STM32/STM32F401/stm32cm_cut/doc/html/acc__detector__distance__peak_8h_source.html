177 /**
178  * @brief Get running average factor
179  *
180  * The running average factor is the factor of which the most recent sweep is weighed against previous sweeps.
181  * Valid range is between 0.0 and 1.0 where 0.0 means that no history is weighed in, i.e filtering is effectively disabled.
182  * A factor of 1.0 means that the most recent sweep has no effect on the result,
183  * which will result in that the first sweep is forever received as the result.
184  * The filtering is coherent and is done on complex valued IQ data before conversion to envelope data.
185  *
186  * @param[in] configuration The configuration to get the running average factor for
187  * @return Running average factor
188  */
189 float acc_detector_distance_peak_running_average_factor_get(acc_detector_distance_peak_configuration_t configuration);

207 /**
208  * @brief Get the sensor downsampling factor
209  *
210  * Gets the downsampling factor - the number of steps taken between each data point. A downsampling factor of 1 samples
211  * every possible point in the range. A downsampling_factor of 2 samples every other point, and so on.
212  *
213  * The base step length is ~0.5mm. Thus, for example setting downsampling_factor to 4 makes the distance
214  * between two points in the measured range ~2mm.
215  *
216  * @param[in] configuration The configuration to get downsampling factor from
217  * @return The downsampling factor
218  */
219 uint16_t acc_detector_distance_peak_downsampling_factor_get(acc_detector_distance_peak_configuration_t configuration);

224  *
225  * Sets the downsampling factor - the number of steps taken between each data point. A downsampling factor of 1 samples
226  * every possible point in the range. A downsampling factor of 2 samples every other point, and so on.
227  *
228  * The base step length is ~0.5mm. Thus, for example setting downsampling_factor to 4 makes the distance

230  *
