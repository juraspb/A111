16 /**
17  * @brief Example that shows how to use the distance peak detector
18  *
19  * This is an example on how the distance peak detector can be used.
20  * The example executes as follows:
21  * - Activate Radar System Software (RSS)
22  * - Create a distance peak detector configuration
23  * - Create a distance peak detector using the previously created configuration
24  * - Activate the distance peak detector
25  * - Get the result and print it 5 times
26  * - Deactivate and destroy the distance peak detector
27  * - Destroy the distance peak detector configuration
28  * - Deactivate Radar System Software (RSS)
29  */

42 bool acc_example_detector_distance_peak(void)
43 {
44  printf("Acconeer software version %s\n", acc_version_get());
45 
46  acc_hal_t hal = acc_hal_integration_get_implementation();
47 
48  if (!acc_rss_activate(&hal))
49  {
50  fprintf(stderr, "acc_rss_activate() failed\n");
51  return false;
52  }
53 
54  acc_detector_distance_peak_configuration_t distance_configuration = acc_detector_distance_peak_configuration_create();
55 
56  if (distance_configuration == NULL)
57  {
58  fprintf(stderr, "acc_detector_distance_peak_configuration_create() failed\n");
59  acc_rss_deactivate();
60  return false;
61  }
62 
63  // Set detector to use a fixed threshold, return absolute values for the amplitude result
64  // and sort the resulting reflections on amplitude
65  acc_detector_distance_peak_set_threshold_mode_fixed(distance_configuration, THRESHOLD);
66  acc_detector_distance_peak_set_absolute_amplitude(distance_configuration, true);
67  acc_detector_distance_peak_set_sort_by_amplitude(distance_configuration, true);
68 
69  acc_sensor_id_t sensor_id = 1;
70  float start_m = 0.2f;
71  float length_m = 0.5f;
72 
73  acc_base_configuration_t base_configuration = acc_detector_distance_peak_get_base_configuration(distance_configuration);
74 
75  acc_base_configuration_sensor_set(base_configuration, sensor_id);
76  acc_base_configuration_requested_start_set(base_configuration, start_m);
77  acc_base_configuration_requested_length_set(base_configuration, length_m);
78 
79  if (!execute_detection(distance_configuration))
80  {
81  acc_detector_distance_peak_configuration_destroy(&distance_configuration);
82  acc_rss_deactivate();
83  return false;
84  }
85 
86  acc_detector_distance_peak_configuration_destroy(&distance_configuration);
87 
88  acc_rss_deactivate();
89 
90  return true;
91 }

94 bool execute_detection(acc_detector_distance_peak_configuration_t distance_configuration)
95 {
96  acc_detector_distance_peak_handle_t handle = acc_detector_distance_peak_create(distance_configuration);
97 
98  if (handle == NULL)
99  {
100  fprintf(stderr, "acc_detector_distance_peak_create() failed\n");
101  return false;
102  }
103 
104  acc_detector_distance_peak_metadata_t metadata;
105  uint16_t reflection_count_max = 5;
106  acc_detector_distance_peak_reflection_t reflections[reflection_count_max];
107 
108  acc_detector_distance_peak_get_metadata(handle, &metadata);
109 
110  printf("Start: %u mm\n", (unsigned int)(metadata.start_m * 1000.0f));
111  printf("Length: %u mm\n", (unsigned int)(metadata.length_m * 1000.0f));
112  printf("Threshold: %u\n", (unsigned int)THRESHOLD);
113  printf("\n");
114 
115  acc_detector_distance_peak_result_info_t result_info;
116 
117  if (!acc_detector_distance_peak_activate(handle))
118  {
119  fprintf(stderr, "acc_detector_distance_peak_activate() failed\n");
120  acc_detector_distance_peak_destroy(&handle);
121  return false;
122  }
123 
124  bool success = true;
125  const int iterations = 5;
126 
127  for (int i = 0; i < iterations; i++)
128  {
129  uint16_t reflection_count = reflection_count_max;
130 
131  success = acc_detector_distance_peak_get_next(handle,
132  reflections,
133  &reflection_count,
134  &result_info);
135 
136  if (!success)
137  {
138  fprintf(stderr, "acc_detector_distance_peak_get_next() failed\n");
139  break;
140  }
141 
142  print_distances(reflection_count, reflections);
143  }
144 
145  bool deactivated = acc_detector_distance_peak_deactivate(handle);
146 
147  acc_detector_distance_peak_destroy(&handle);
148 
149  return deactivated && success;
150 }

153 void print_distances(uint16_t reflection_count,
154  const acc_detector_distance_peak_reflection_t *reflections)
155 {
156  if (reflection_count == 0)
157  {
158  printf("No peaks above threshold\n");
159  }
160  else
161  {
162  for (uint16_t i = 0; i < reflection_count; i++)
163  {
164  unsigned int distance = (unsigned int)((reflections[i].distance * 1000.0f) + 0.5f);
165  unsigned int amplitude = (unsigned int)(reflections[i].amplitude);
166 
167  printf("Peak %d, Distance: %3u mm, Amplitude: %u\n", i + 1, distance, amplitude);
168  }
169  }
170 
171  printf("\n");
172 }
