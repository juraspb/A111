122 static void acc_hal_integration_sensor_power_on(acc_sensor_id_t sensor_id)
123 {
124  (void)sensor_id; // Ignore parameter sensor_id
125 
126  if (sensor_state == SENSOR_STATE_BUSY)
127  {
128  return;
129  }
130 
131  HAL_GPIO_WritePin(XC111_SPI_SS1_GPIO_Port, XC111_SPI_SS1_Pin, GPIO_PIN_RESET);
132  HAL_GPIO_WritePin(XC111_SPI_SS0_GPIO_Port, XC111_SPI_SS0_Pin, GPIO_PIN_RESET);
133 
134  if (acc_hal_integration_all_sensors_inactive())
135  {
136  HAL_GPIO_WritePin(A111_RESET_N_GPIO_Port, A111_RESET_N_Pin, GPIO_PIN_RESET);
137 
138  HAL_GPIO_WritePin(XC111_PMU_ENABLE_GPIO_Port, XC111_PMU_ENABLE_Pin, GPIO_PIN_SET);
139 
140  // Wait for PMU to stabilize
141  acc_hal_integration_sleep_us(5000);
142 
143  HAL_GPIO_WritePin(A111_ENABLE_GPIO_Port, A111_ENABLE_Pin, GPIO_PIN_SET);
144 
145  // Wait for Power On Reset
146  acc_hal_integration_sleep_us(5000);
147  HAL_GPIO_WritePin(A111_SPI_SS_GPIO_Port, A111_SPI_SS_Pin, GPIO_PIN_SET);
148  HAL_GPIO_WritePin(A111_RESET_N_GPIO_Port, A111_RESET_N_Pin, GPIO_PIN_SET);
149 
150  sensor_state = SENSOR_STATE_READY;
151  }
152 
153  if (sensor_state != SENSOR_STATE_READY)
154  {
155  return;
156  }
157 
158  sensor_state = SENSOR_STATE_BUSY;
159 
160  return;
161 }

164 static void acc_hal_integration_sensor_power_off(acc_sensor_id_t sensor_id)
165 {
166  (void)sensor_id; // Ignore parameter sensor_id
167 
168  bool status;
169 
170  if (sensor_state != SENSOR_STATE_BUSY)
171  {
172  return;
173  }
174 
175  status = acc_hal_integration_reset_sensor();
176 
177  if (status)
178  {
179  sensor_state = SENSOR_STATE_UNKNOWN;
180  }
181 
182  return;
183 }

186 static bool acc_hal_integration_wait_for_sensor_interrupt(acc_sensor_id_t sensor_id, uint32_t timeout_ms)
187 {
188  (void)sensor_id; // Ignore parameter sensor_id
189 
190  HAL_GPIO_WritePin(XC111_SPI_SS1_GPIO_Port, XC111_SPI_SS1_Pin, GPIO_PIN_RESET);
191  HAL_GPIO_WritePin(XC111_SPI_SS0_GPIO_Port, XC111_SPI_SS0_Pin, GPIO_PIN_RESET);
192 
193  const uint32_t wait_begin_ms = HAL_GetTick();
194  while (HAL_GPIO_ReadPin(A111_SENSOR_INTERRUPT_GPIO_Port, A111_SENSOR_INTERRUPT_Pin) != GPIO_PIN_SET)
195  {
196  if (timeout_ms == 0 || HAL_GetTick() - wait_begin_ms > timeout_ms)
197  {
198  return false;
199  }
200 
201  // Wait for the next tick interrupt
202  __WFI();
203  }
204 
205  return true;
206 }

221 static void acc_hal_integration_log(acc_log_level_t level, const char *module, const char *buffer)
222 {
223  uint32_t time_ms = acc_hal_integration_get_current_time();
224  char level_ch;
225 
226  unsigned int timestamp = time_ms;
227  unsigned int hours = timestamp / 1000 / 60 / 60;
228  unsigned int minutes = timestamp / 1000 / 60 % 60;
229  unsigned int seconds = timestamp / 1000 % 60;
230  unsigned int milliseconds = timestamp % 1000;
231 
232  level_ch = (level <= ACC_LOG_LEVEL_DEBUG) ? "EWIVD"[level] : '?';
233 
234  printf(LOG_FORMAT, hours, minutes, seconds, milliseconds, 0, level_ch, module, buffer);
235 }
